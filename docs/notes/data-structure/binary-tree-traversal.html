var levelOrder = function(root) {
    if (root === null || root.length === 0) {
        return [];
    }

    const result = [];
    const waitingList = [ { level: 0, node: root } ];

    while (waitingList.length > 0) {
        // è™•ç† waiting list ç•¶å‰çš„ç¯€é»

        }
    }

    return result;
};
